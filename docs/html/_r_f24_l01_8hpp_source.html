1112  }

1116  write_command(COMMAND::ACTIVATE);
1117  }
1118  setting_enable(SETTING::EN_DYN_ACK);
1119  write(COMMAND::W_TX_PAYLOAD_NO_ACK, Data);
1120  setting_disable(SETTING::EN_DYN_ACK);
1121  if(!prev_active) {
1122  write_command(COMMAND::ACTIVATE);
1123  }
1124  } else {
